# Manual tests for the caching proxy server (Lab 5)
#
# Start the server in another terminal:
#   npm run dev -- -h 127.0.0.1 -p 8080 -c cache

# --- GET: cache miss → proxy fetch from http.cat, then save to cache ---
# Start with an empty cache (rm -rf cache && mkdir cache).
curl -s -o /tmp/got-200.jpg -w "%{http_code} %{content_type} size=%{size_download}\n" http://127.0.0.1:8080/200
# Expected: 200 image/jpeg, server log: "Cache miss: 200" then "Saved to cache: 200".
# After this call, cache/200.jpg exists.

# --- GET: cache hit on the second call (no http.cat request) ---
curl -s -o /tmp/got-200-again.jpg -w "%{http_code} %{content_type} size=%{size_download}\n" http://127.0.0.1:8080/200
# Expected: 200 image/jpeg, server log: "Cache hit: 200".
# /tmp/got-200.jpg and /tmp/got-200-again.jpg must be byte-identical:
cmp -s /tmp/got-200.jpg /tmp/got-200-again.jpg && echo "identical" || echo "different"

# --- GET: code that does not exist on http.cat → 404 ---
curl -i http://127.0.0.1:8080/999
# Expected: HTTP/1.1 404 Not Found, body: Not Found

# --- Validation: empty / non-numeric code ---
curl -i http://127.0.0.1:8080/
curl -i http://127.0.0.1:8080/abc
# Expected: HTTP/1.1 400 Bad Request

# --- PUT: upload a jpg into the cache ---
# Take any local jpg (e.g. cat.jpg) and upload it under code 200:
curl -i -X PUT --data-binary "@cat.jpg" http://127.0.0.1:8080/200
# Expected: HTTP/1.1 201 Created, body: Created
# Verify the upload by reading it back through GET:
curl -s -o /tmp/got-200.jpg -w "%{http_code} %{content_type}\n" http://127.0.0.1:8080/200
# Expected: 200 image/jpeg, and /tmp/got-200.jpg matches cat.jpg byte-for-byte.

# --- DELETE: remove an existing cache file ---
# Make sure the file exists first (PUT it, see above), then:
curl -i -X DELETE http://127.0.0.1:8080/200
# Expected: HTTP/1.1 200 OK, body: OK
# A second DELETE for the same code should now return 404:
curl -i -X DELETE http://127.0.0.1:8080/200
# Expected: HTTP/1.1 404 Not Found, body: Not Found

# --- DELETE: missing cache file ---
curl -i -X DELETE http://127.0.0.1:8080/777
# Expected: HTTP/1.1 404 Not Found, body: Not Found

# --- Method not allowed (only GET, PUT, DELETE are implemented) ---
curl -i -X POST http://127.0.0.1:8080/200
# Expected: HTTP/1.1 405 Method Not Allowed
